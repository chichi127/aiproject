from flask import Flask, render_template, request
import os
import torch
import cv2
from torchvision import transforms
from PIL import Image
from collections import Counter
from model import SimpleCNN
from utils import extract_frames, predict_single_image

app = Flask(__name__)
UPLOAD_FOLDER = 'uploads'
os.makedirs(UPLOAD_FOLDER, exist_ok=True)

label_map = {
    0: "happiness",
    1: "disgust",
    2: "repression",
    3: "sadness",
    4: "surprise"
}

emoji_map = {
    0: "ğŸ˜„",  # happiness
    1: "ğŸ¤¢",  # disgust
    2: "ğŸ˜¶",  # repression (ì–µì œëœ ê°ì • â†’ ì¤‘ë¦½)
    3: "ğŸ˜¢",  # sadness
    4: "ğŸ˜²"   # surprise
}


# ì´ë¯¸ì§€ ì „ì²˜ë¦¬
transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.5]*3, std=[0.5]*3)
])

# ëª¨ë¸ ë¶ˆëŸ¬ì˜¤ê¸°
model = SimpleCNN(num_classes=5)
model.load_state_dict(torch.load('SimpleCNN_model.pth', map_location='cpu'))
model.eval()

# ğŸ”½ ì—¬ê¸°ì— index() í•¨ìˆ˜ì™€ í•¨ê»˜ predict() í•¨ìˆ˜ë„ ì´ì–´ì„œ ë“¤ì–´ê°
@app.route('/')
def index():
    return render_template('index.html')

# âœ… ì´ ìë¦¬ì— ë„¤ê°€ ì“´ predict() í•¨ìˆ˜ ì „ì²´ ë¶™ì—¬ë„£ê¸°!
@app.route('/predict', methods=['POST'])
def predict():
    file = request.files['file']
    if file:
        filepath = os.path.join(UPLOAD_FOLDER, file.filename)
        file.save(filepath)

        # 1. í”„ë ˆì„ ì¶”ì¶œ
        frames = extract_frames(filepath, frame_skip=10)

        # 2. í”„ë ˆì„ë³„ ì˜ˆì¸¡ ìˆ˜í–‰
        predictions = [predict_single_image(model, frame, transform) for frame in frames]

        # 3. ì˜ˆì¸¡ ê²°ê³¼ ì²˜ë¦¬
        if predictions:
            final_pred = Counter(predictions).most_common(1)[0][0]  # ğŸ‘ˆ ì—¬ê¸°!
            label = label_map[final_pred]                           # ğŸ‘ˆ ê°ì • ì´ë¦„
            emoji = emoji_map[final_pred]                           # ğŸ‘ˆ ê°ì • ì´ëª¨ì§€

            # âœ… ëŒ€í‘œ í”„ë ˆì„ ì €ì¥ (ê°€ì¥ ì²˜ìŒ ì˜ˆì¸¡í•œ í”„ë ˆì„ ì‚¬ìš©)
            preview_frame = frames[0]  # ë˜ëŠ” final_predì™€ ì¼ì¹˜í•œ í”„ë ˆì„ ì¤‘ í•˜ë‚˜ ì„ íƒ
            preview_path = os.path.join("static", "preview.jpg")
            Image.fromarray(cv2.cvtColor(preview_frame, cv2.COLOR_BGR2RGB)).save(preview_path)

        else:
            label = "prediction failed"
            emoji = "âŒ"
            preview_path = None

        # 4. ê²°ê³¼ í˜ì´ì§€ë¡œ ë„˜ê¹€
        return render_template('result.html', label=label, emoji=emoji, preview_image='preview.jpg')

    return render_template('index.html', result="Upload failed")


# Flask ì•± ì‹¤í–‰ ë¶€ë¶„ì€ ë§¨ ì•„ë˜
if __name__ == '__main__':
    app.run(debug=True)
